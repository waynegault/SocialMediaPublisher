#   pip install -r requirements-z-image.txt

# Requires NVIDIA GPU with CUDA support:
#   - 4GB VRAM: Works with sequential CPU offload (~50GB system RAM needed)
#   - 8GB+ VRAM: Works with model CPU offload
#   - 16GB+ VRAM: Full GPU acceleration
#
# Configure in .env:
#   IMAGE_PROVIDER=z_image
#   Z_IMAGE_DTYPE=bfloat16   # Use bfloat16 on Ampere+ GPUs to avoid NaN issues
#   Z_IMAGE_OFFLOAD=1        # Enable CPU offloading for low-VRAM GPUs
#   Z_IMAGE_SIZE=512x512     # Recommended for 4GB VRAM
#   Z_IMAGE_STEPS=20         # Fewer steps = faster on low VRAM
